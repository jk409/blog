镜像下载

#版本：
postgres:[12.4/13]
odoo:[14/15]

#docekr pull docker.io/library/postgres:12.4
dcoker pull postgres:12.4
dcoker pull odoo:14

运行postgres

docker run -d -p 5432:5432  --name db \
    -e POSTGRES_DB=postgres \    #该项不能修改变动，必须为postgres
    -e POSTGRES_USER=odoo \
    -e POSTGRES_PASSWORD=odoo \
    -v /opt/postgresql/data:/var/lib/postgresql/data
    postgres:12.4

运行odoo

docker run -d -p 8069:8069 --name odoo \
    --link db:db \
    -v /opt/odoo:/etc/odoo \
    -v /opt/odoo_addons:/mnt/extra-addons \
    -t odoo:14

后台登录
