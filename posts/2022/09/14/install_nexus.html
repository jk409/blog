文章目录

下载镜像

docker pull registry.cn-qingdao.aliyuncs.com/kubeoperator/nexus3:3.25.0-amd64
或者
docker pull sonatype/nexus3

运行

#docker run -d --privileged --restart=unless-stopped \
-p 80:80 -p 443:443 rancher/rancher:v2.5.12

#sudo docker run -d --restart=always  \
    --privileged \
    -p 8081:8081 \
    -p 8082:8082 \
    -p 808:8083 \
    -v /opt/nexus/:/nexus-data \
    --name=nexus  \
    registry.cn-qingdao.aliyuncs.com/kubeoperator/nexus3:3.25.0-amd64

#cat /data/nexus/data/admin.password

docker-compose

#cat > docker-compose.yml <<EOF
version: "1.0"
services:
    nexus:
        restart: always
        image: registry.cn-qingdao.aliyuncs.com/kubeoperator/nexus3:3.25.0-amd64
        container_name: nexus
        ports:
        - 8081:8081
        - 8082:8082
        - 8083:8083
        volumes:
        - /opt/nexus/:/nexus-data
        healthcheck:
        test: ["CMD","curl","localhost:8081"]
        interval: 10s
        timeout: 10s
        retries: 20
EOF

#docker-compose up -d

Nexus 后台管理

localhost:8081

localhost:8081/nexus

admin admin123 或 #密码是在挂载目录下的admin.password文件中，/opt/nexus/admin.password

Linux客户端配置repo

centos 配置repo

#mkdir /etc/yum.repos.d/bak
#mv /etc/yum.repos.d/*.repo  /etc/yum.repos.d/bak
#cat /etc/yum.repos.d/kubeops.repo
[Centos-Base]
name=CentOS Base
baseurl=http://172.20.2.90:8081/repository/centos-base/7/extras/$basearch/
enabled=1
gpgcheck=0
protect=1


[Centos-Extras]
name=CentOS Extras
baseurl=http://172.20.2.90:8081/repository/centos-base/7/os/$basearch/
enabled=1
gpgcheck=0
protect=1

[Centos-epel]
name=CentOS epel
baseurl=http://172.20.2.90:8081/repository/centos-epel/7/$basearch/
enabled=1
gpgcheck=0
protect=1

debian 10 buster 配置repo

sudo cp /etc/apt/sources.list /etc/apt/sources.list.backup 
# 默认注释了源码镜像以提高 apt update 速度，如有需要可自行取消注释
deb https://mirrors.tuna.tsinghua.edu.cn/debian/ buster main contrib non-free
# deb-src https://mirrors.tuna.tsinghua.edu.cn/debian/ buster main contrib non-free
deb https://mirrors.tuna.tsinghua.edu.cn/debian/ buster-updates main contrib non-free
# deb-src https://mirrors.tuna.tsinghua.edu.cn/debian/ buster-updates main contrib non-free
deb https://mirrors.tuna.tsinghua.edu.cn/debian/ buster-backports main contrib non-free
# deb-src https://mirrors.tuna.tsinghua.edu.cn/debian/ buster-backports main contrib non-free
deb https://mirrors.tuna.tsinghua.edu.cn/debian-security buster/updates main contrib non-free
# deb-src https://mirrors.tuna.tsinghua.edu.cn/debian-security buster/updates main contrib non-free

树莓派配置repo

#清华
sudo sed -i 's#://mirrordirector.raspbian.org#s://mirrors.tuna.tsinghua.edu.cn/raspbian#g' /etc/apt/sources.list
sudo sed -i 's#://archive.raspberrypi.org/debian#s://mirrors.tuna.tsinghua.edu.cn/raspberrypi#g' /etc/apt/sources.list.d/raspi.list



#私有仓库（http 和 https）
sudo sed -i 's#://mirrordirector.raspbian.org#s://172.20.2.90:8081/repository/raspbian#g' /etc/apt/sources.list
sudo sed -i 's#://archive.raspberrypi.org/debian#s://172.20.2.90:8081/repository/raspberrypi#g' /etc/apt/sources.list.d/raspi.list
